import db from "@/lib/db";
import { NextResponse } from "next/server";

//Receber/Ver
export async function GET() {
    const [rows] = await db.query("SELECT * FROM (nome da tabela)");
    return NextResponse.json(rows);
}

//Criar
export async function POST(req: Request){
    const {nome} = await req.json();
    await db.query(
        "INSERT INTO (nome da tabela)(nome) VALUES(?)",
        [nome]
    );
    return NextResponse.json({
        success:true,
    });
}

//Atualizar ou Modificar algo
export async function PUT(req: Request){

    const {novo_nome,antigo_nome} = await req.json();
    
    await db.query("UPDATE (nome da tabela) SET nome = ? WHERE nome = ?",
        [novo_nome,antigo_nome]
    );
    return NextResponse.json({
        success: true,
    });

}
//Deletar/excluir
export async function DELETE(req: Request){
    const {nome} = await req.json();
    await db.query("DELETE FROM (nome da tabela) WHERE nome = ?",
        [nome]
    );
    return NextResponse.json({
        success: true,
    });
}
